📊 Sync ke Google Sheets
Cara kerja: Sistem MIARS hantar data artikel terus ke Google Sheets awak secara automatik melalui Google Apps Script (percuma).
1. Setup
2. URL
3. Sync!
Langkah 1 — Setup Google Apps Script
1. Buka sheets.google.com → buat spreadsheet baharu
2. Namakan: MIARS Literature Database
3. Klik Extensions → Apps Script
4. Padam semua teks dalam editor
5. Salin kod di bawah → tampal → klik 💾 Save
6. Klik Deploy → New deployment
7. Type: Web app → Execute as: Me → Who has access: Anyone
8. Klik Deploy → Authorize access → salin URL yang dijana
function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var articles = data.articles || []; // Sheet 1: Database var sheet = ss.getSheetByName("Database") || ss.insertSheet("Database"); sheet.clearContents(); var headers = ["Kod","Tema","Penulis","Tahun","Tajuk","Jurnal","Vol","DOI", "Negara","Jenis","Quartile","Tahap","PDF","Sampel","Dapatan","Gap", "Hubungan MIARS","Ayat Tesis","Bab","RO","Topik","Nota","GDrive"]; sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.getRange(1, 1, 1, headers.length).setBackground("#1F4E79").setFontColor("#FFFFFF").setFontWeight("bold"); if (articles.length > 0) { var rows = articles.map(function(a) { return [ a.kod||"", a.tema||"", a.penulis||"", a.tahun||"", a.tajuk||"", a.jurnal||"", a.vol||"", a.doi||"", a.negara||"", a.jenis||"", a.quartile||"", a.tahapArtikel||"", a.pdf||"", a.sampel||"", (a.dapatan||[]).join(" | "), (a.gap||[]).join(" | "), a.miars||"", a.ayat||"", a.bab||"", a.ro||"", a.topik||"", a.nota||"", a.gdrive||"" ]; }); sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); sheet.autoResizeColumns(1, headers.length); } // Sheet 2: Summary var sum = ss.getSheetByName("Summary") || ss.insertSheet("Summary"); sum.clearContents(); sum.getRange("A1").setValue("MIARS Literature Database Summary"); sum.getRange("A1").setFontSize(14).setFontWeight("bold").setFontColor("#1F4E79"); sum.getRange("A2").setValue("Dikemas kini: " + new Date().toLocaleString()); sum.getRange("A2").setFontColor("#666666"); sum.getRange("A4").setValue("Jumlah Artikel").setFontWeight("bold"); sum.getRange("B4").setValue(articles.length); var myCount = articles.filter(function(a){return a.negara==="Malaysia";}).length; sum.getRange("A5").setValue("Artikel Malaysia").setFontWeight("bold"); sum.getRange("B5").setValue(myCount); var q1 = articles.filter(function(a){return a.quartile==="Q1";}).length; sum.getRange("A6").setValue("Artikel Q1").setFontWeight("bold"); sum.getRange("B6").setValue(q1); sum.getRange("A8").setValue("Tema").setFontWeight("bold").setBackground("#E8F5E9"); sum.getRange("B8").setValue("Bilangan").setFontWeight("bold").setBackground("#E8F5E9"); var themes = ["T1","T2","T3","T4","T5","T6","T7","T8","T9","T10"]; var tNames = ["Penguasaan Sejarah","Murid Berprestasi Rendah","Pembelajaran Inkuiri", "Adaptive Learning","RAG/AI","Scaffolding","AI Pendidikan","DDR & Model", "Penilaian","Sejarah & Teknologi"]; themes.forEach(function(t, i) { var n = articles.filter(function(a){return a.tema===t;}).length; sum.getRange(9+i, 1).setValue(t + " - " + tNames[i]); sum.getRange(9+i, 2).setValue(n); }); sum.autoResizeColumns(1, 2); return ContentService.createTextOutput(JSON.stringify({ success: true, message: articles.length + " artikel berjaya disimpan ke Google Sheets!", timestamp: new Date().toISOString() })).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Ralat: " + err.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: "MIARS Sync aktif", timestamp: new Date().toISOString() })).setMimeType(ContentService.MimeType.JSON); }
